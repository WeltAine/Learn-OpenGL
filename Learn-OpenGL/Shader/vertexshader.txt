#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec3 aColour;
layout (location = 2) in vec2 aUV;

out vec3 colour;
out vec2 uv;

uniform float xOffset;
uniform mat4 transform;

void main(){
    colour = aColour;
    gl_Position = transform * vec4(aPos.x + xOffset, aPos.y, aPos.z, 1.0f);
    uv = aUV;
}
